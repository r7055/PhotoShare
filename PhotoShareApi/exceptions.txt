try
{
    // קוד שיכול לזרוק שגיאה
}
catch (BadRequestException ex)
{
    return StatusCode(400, $"בקשה לא חוקית: {ex.Message}");
}
catch (UnauthorizedAccessException ex)
{
    return StatusCode(401, $"גישה לא מורשית: {ex.Message}");
}
catch (ForbiddenException ex)
{
    return StatusCode(403, $"גישה אסורה: {ex.Message}");
}
catch (NotFoundException ex)
{
    return StatusCode(404, $"לא נמצא: {ex.Message}");
}
catch (Exception ex)
{
    return StatusCode(500, $"שגיאה לא צפויה: {ex.Message}");
}
